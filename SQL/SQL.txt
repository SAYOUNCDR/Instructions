------ Working with Columns ---------

# Edit table i.e add new col
ALTER TABLE <table_name> ADD <col_name> <col_datatype>

# Rename a column
ALTER TABLE <table_name> RENAME COLUMN <old_colname> TO <new_colname>

# Chnage the datatype for a Column 
ALTER TABLE <table_name> MODIFY COLUMN <col_name_who_datatype_needs _to_change> <new_datatype>


# Move the position of a column in the table after a particular col
ALTER TABLE <table_name> MODIFY <col_name_which_needs_to_shift> <its_datatype> AFTER <col_name_after_which_the_col_wants_to_be_moved>

# Move the col to as first col

# Delete a column
ALTER TABLE <table_name> DROP COLUMN <col_name>

# Tf is Auto-Commit?
Auto-Commit is a feature in SQL that automatically saves changes to the database after each individual statement is executed. 
This means that once a statement is executed, the changes are immediately permanent and cannot be rolled back.

# Why is Auto-Commit important? 
Auto-Commit is important because it ensures that changes to the database are saved immediately, which can help prevent data loss in case of unexpected errors or crashes.
However, it can also lead to unintended consequences if not used carefully, as it may commit changes that were not intended to be permanent.


# How to disable Auto-Commit?
To disable Auto-Commit, you can use the following command:
SET autocommit = OFF;

# How to enable Auto-Commit?
To enable Auto-Commit, you can use the following command:
SET autocommit = ON;


# What is a Transaction?
A transaction is a sequence of one or more SQL statements that are executed as a single unit of work.



-- To save a commit point 
COMMIT;

-- To rollback to a commit point
ROLLBACK;

Note- If auto-commit is disabled every transaction will be pending until commit

-- Dates and time in SQL

# Date time and datetime datatype
DATE- stores only date in the format 'YYYY-MM-DD'
DATETIME- stores both date and time in the format 'YYYY-MM-DD HH:MM:
TIME- stores only time in the format 'HH:MM:SS'


example-
CREATE TABLE events (
	id INT,
	event_name VARCHAR(50),
	event_date DATE,
	event_time TIME,
	event_datetime DATETIME
)

# To insert date and time data into a table
INSERT INTO events VALUES (
	1, 'Birthday Party', '2025-01-15', '18:00:00', '2025-01-15 18:00:00'
)

# Some date and time functions in SQL
CURRENT_DATE() - returns the current date
CURRENT_TIME() - returns the current time
CURRENT_TIMESTAMP() or NOW() - returns the current date and time


example-

INSERT INTO events VALUES (
	2, 'Conference', CURRENT_DATE(), CURRENT_TIME(), NOW()
)
